<body>
    <div class="container">
        <h1 class="mt-5">Hello from ELFInsight Webview!</h1>
        <button id="loadSymbolsButton" class="btn btn-primary mt-3">Load Symbols</button>
        <table class="table mt-4">
            <thead>
                <tr>
                    <th>Address</th>
                    <th>Size</th>
                    <th>Type</th>
                    <th>Name</th>
                    <th>Section</th>
                </tr>
            </thead>
            <tbody id="symbolsTable">
                <!-- Symbol rows will be added here dynamically -->
            </tbody>
        </table>
    </div>
    <script>
        (function() {
            const vscode = acquireVsCodeApi();

            document.getElementById('loadSymbolsButton').addEventListener('click', () => {
                console.log('Load Symbols button clicked');
                vscode.postMessage({ command: 'loadSymbols' });
            });

            window.addEventListener('message', event => {
                const message = event.data;
                console.log('Received message from extension:', message);
                if (message.command === 'displaySymbols') {
                    const symbolsTable = document.getElementById('symbolsTable');
                    symbolsTable.innerHTML = '';
                    message.symbols.forEach(symbol => {
                        const row = document.createElement('tr');
                        row.innerHTML = `
                            <td>${symbol.address}</td>
                            <td>${symbol.size}</td>
                            <td>${symbol.type}</td>
                            <td>${symbol.name}</td>
                            <td>${symbol.section}</td>
                        `;
                        symbolsTable.appendChild(row);
                    });
                }
            });
        }());
    </script>
